Quick Start

        
# 1. Get an API key from the API Keys page
# 2. Run synchronous inference

curl -X POST https://your-meshpilot.host/api/v1/inference/sync/{model_id} \
  -H "X-API-Key: mp_your_key_here" \
  -H "Content-Type: application/json" \
  -d '{
    "prompt": "Explain CPU-only AI inference in one sentence.",
    "max_tokens": 256,
    "temperature": 0.7
  }'

# Response:
{
  "id": "abc123",
  "status": "completed",
  "text": "CPU-only AI inference runs large language models...",
  "latency_ms": 2840.5,
  "throughput_tps": 18.4,
  "backend_used": "llamacpp"
}
        

        
Async Inference

        
# Submit async job
curl -X POST .../api/v1/inference/async/{model_id} \
  -H "X-API-Key: mp_..." \
  -d '{"prompt": "...", "webhook_url": "https://your-server/callback"}'

# Returns immediately:
{"task_id": "xyz789", "status": "pending", "poll_url": "/api/v1/inference/tasks/xyz789"}

# Poll for result
curl .../api/v1/inference/tasks/xyz789 -H "X-API-Key: mp_..."
        

        
Model Upload

        
# Upload a GGUF model (auto-quantized to INT4)
curl -X POST .../api/v1/models \
  -H "Authorization: Bearer {jwt_token}" \
  -F "file=@./my-model.gguf" \
  -F "name=My Custom Model" \
  -F "description=Fine-tuned on domain data"

# Check quantization status
curl .../api/v1/models/{model_id} -H "Authorization: Bearer {jwt_token}"
